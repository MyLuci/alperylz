<!-- CUSTOM CURSOR TEMPLATE - Copy this code for all new HTML files -->

<!-- Add this CSS in your <style> section -->
<style>
/* Custom Cursor - INCLUDE THIS IN ALL FUTURE HTML FILES */
* {
    cursor: none !important;
}

body {
    cursor: none !important;
}

.custom-cursor {
    position: fixed;
    top: 0;
    left: 0;
    width: 40px;
    height: 40px;
    border: 2px solid var(--bauhaus-red);
    border-radius: 50%;
    pointer-events: none;
    z-index: 10000;
    transition: transform 0.1s ease, border-color 0.3s ease;
    transform: translate(-50%, -50%);
}

.custom-cursor.hover {
    transform: translate(-50%, -50%) scale(1.5);
    border-color: var(--bauhaus-blue);
    border-radius: 0;
}

.custom-cursor.click {
    transform: translate(-50%, -50%) scale(0.8);
    border-color: var(--bauhaus-yellow);
}
</style>

<!-- Add this HTML right after <body> tag -->
<body>
    <!-- Custom Cursor -->
    <div class="custom-cursor"></div>
    
    <!-- Your content starts here... -->

<!-- Add this JavaScript before </body> tag -->
<script>
// Custom Cursor
const cursor = document.querySelector('.custom-cursor');

// Move cursor with mouse
document.addEventListener('mousemove', (e) => {
    cursor.style.left = e.clientX + 'px';
    cursor.style.top = e.clientY + 'px';
});

// Add hover effects
const hoverElements = document.querySelectorAll('a, button, .project-card, .skill-item, .contact-link');
hoverElements.forEach(element => {
    element.addEventListener('mouseenter', () => {
        cursor.classList.add('hover');
    });
    element.addEventListener('mouseleave', () => {
        cursor.classList.remove('hover');
    });
});

// Add click effect
document.addEventListener('mousedown', () => {
    cursor.classList.add('click');
});
document.addEventListener('mouseup', () => {
    cursor.classList.remove('click');
});

// Hide cursor when leaving window
document.addEventListener('mouseleave', () => {
    cursor.style.display = 'none';
});
document.addEventListener('mouseenter', () => {
    cursor.style.display = 'block';
});
</script>
</body>
